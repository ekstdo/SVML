import numpy as np
import cv2


def lin_map(x, a, b, A, B):                 # define linear mapping function
    return (x-a)*(B-A)/(b-a) + A


# Implementieren Sie die Funktion CoordZuZahl, die einer Pixel-Koordinate eine
# komplexe Zahl zuweist
def CoordZuZahl(CoordX, CoordY, bildgroesse, grenzen_reell, grenzen_im):
    Re = lin_map(float(CoordX), 0., float(bildgroesse[0] - 1), float(grenzen_reell[0]), float(grenzen_reell[1]))
    # wichtig: die y-Achse ist bei einem Bild vertauscht zur Koordinatenachse!
    # Daher wird y_max - y berechnet:
    Im = lin_map(float(bildgroesse[1] - 1) - float(CoordY), 0., float(bildgroesse[1] - 1), float(grenzen_im[0]), float(grenzen_im[1]))
    return complex(Re, Im)

# Weisen Sie jeder Pixel-Koordinate im Mandelbrot-Bild eine komplexe Zahl zu und
# überprüfen Sie, ob diese Zahl divergiert.
def mandelbrot(groesse=[200, 150],
               grenzen_reell=(-2., 1.),
               grenzen_im=(-1.1, 1.1),
               pruef_iterationen=50):
    ergebnis = np.zeros(groesse[::-1]+[3], dtype='uint8')
  
    # Sie sind dran!
    for x in range(groesse[0]):
        for y in range(groesse[1]):
            z = CoordZuZahl(x, y, groesse, grenzen_reell, grenzen_im)
            s = complex(0., 0.)
            iter_num = pruef_iterationen
            for i in range(pruef_iterationen):
                s = s ** 2 + z
                if abs(s) > 2:
                    iter_num = i
                    break
            ergebnis[y, x, 0] = lin_map(iter_num, 0., pruef_iterationen, 0., 255.) # blau wert in imshow
  
    return ergebnis
cv2.imshow("Mandelbrot Menge", mandelbrot([200, 150]))
cv2.waitKey()
cv2.destroyAllWindows()
